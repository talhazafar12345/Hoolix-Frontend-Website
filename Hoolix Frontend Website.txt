export default function Print() {

  const [menuOpen, setMenuOpen] = useState(false);
  const [scrolled, setScrolled] = useState();

  const toggleMenu = () => setMenuOpen(!menuOpen);

  useEffect(() => {

    const handleScroll = () => {
      if (window.scrollY) {
        setScrolled(!scrolled);
      } else {
        setScrolled(scrolled);
      }
    };

    window.addEventListener("scroll", handleScroll)

  }, []);


  return (
    <div>


      <Router>
        <div className={`nav ${scrolled ? "scrolled" : ""}`}>
          <h1>Hoolix</h1>

          <div className={`navlink ${menuOpen ? "open" : ""}`}>
            <NavLink className="link" to="/">Home</NavLink>
            <NavLink className="link" to="/features">About</NavLink>
            <NavLink className="link" to="/about">Services</NavLink>
            <NavLink className="link" to="/app">Features</NavLink>
            <NavLink className="link" to="/blog">Contact</NavLink>
            <NavLink className="link" to="/contact">Blog</NavLink>
            <button>Lets Talk</button>
          </div>

          <div className="hamburger" onClick={toggleMenu}>
            {menuOpen ? "✖" : "☰"}
          </div>

        </div>

        <Routes>
          <Route path="/" />
          <Route path="/features" />
          <Route path="/about" />
          <Route path="/app" />

  )
}
